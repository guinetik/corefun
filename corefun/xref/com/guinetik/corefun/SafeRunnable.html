3   /**
4    * A functional interface for operations that may throw checked exceptions.
5    * <p>
6    * {@code SafeRunnable} allows writing lambda expressions for operations that throw
7    * checked exceptions, which can then be converted to standard {@link Runnable}
8    * instances with automatic exception wrapping. This eliminates the boilerplate
9    * of try-catch blocks in lambda expressions.
10   * </p>
11   *
12   * <h2>Problem Solved</h2>
13   * <p>
14   * Java's {@link Runnable} interface declares {@code void run()} without any
15   * checked exceptions. This makes it impossible to use lambdas that throw checked
16   * exceptions directly with executors or other APIs expecting Runnable.
17   * </p>
18   *
19   * <h2>Example Usage</h2>
20   * <pre class="language-java"><code>
21   * // Define an operation that throws checked exceptions
22   * SafeRunnable task = () -&gt; {
23   *     Files.delete(path);  // throws IOException
24   * };
25   *
26   * // Convert to standard Runnable for use with executors
27   * executor.execute(task.toRunnable());
28   *
29   * // Or execute immediately with automatic wrapping
30   * task.execute();  // throws SafeException on failure
31   *
32   * // Create from existing Runnable
33   * SafeRunnable fromRunnable = SafeRunnable.from(() -&gt; doSomething());
34   * </code></pre>
35   *
36   * @author Guinetik &lt;guinetik@gmail.com&gt;
37   * @since 0.1.0
38   * @see SafeCallable
39   * @see SafeException
40   * @see SafeExecutor#safelySafe(String, SafeRunnable)
41   */
42  @FunctionalInterface
43  public interface SafeRunnable {
44  
45      /**
46       * Executes the operation, potentially throwing a checked exception.
47       *
48       * @throws Exception if the operation fails
49       */
50      void run() throws Exception;
51  
52      /**
53       * Converts this SafeRunnable to a standard Runnable.
54       * Any checked exceptions are wrapped in {@link SafeException}.
55       *
56       * @return a Runnable that wraps checked exceptions
57       */
58      default Runnable toRunnable() {
59          return () -> {
60              try {
61                  run();
62              } catch (Exception e) {
63                  throw SafeException.wrap(e);
64              }
65          };
66      }
67  
68      /**
69       * Creates a SafeRunnable from a standard Runnable.
70       *
71       * @param runnable the Runnable to wrap
72       * @return a SafeRunnable that delegates to the Runnable
73       */
74      static SafeRunnable from(Runnable runnable) {
75          return runnable::run;
76      }
77  
78      /**
79       * Executes this SafeRunnable, wrapping any exception.
80       * Convenience method for immediate execution.
81       *
82       * @throws SafeException if execution fails
83       */
84      default void execute() {
85          toRunnable().run();
86      }
87  }
